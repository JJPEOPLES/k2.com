# Task Manager Application using Talanolius GUI
import Talanolius

# Create the application
app = Talanolius.Application.new("Task Manager")

# Create the main window
window = Talanolius.Window.new({
  title: "Talanolius Task Manager",
  width: 1000,
  height: 700,
  resizable: true
})

# Create the layout
layout = Talanolius.HBox.new()

# Create the sidebar
sidebar = Talanolius.VBox.new({
  width: 250,
  backgroundColor: "#f5f5f5",
  borderRight: "1px solid #e0e0e0"
})

# Add sidebar items
sidebarItems = [
  { icon: "home", text: "Dashboard", active: true },
  { icon: "tasks", text: "Tasks" },
  { icon: "calendar", text: "Calendar" },
  { icon: "chart-bar", text: "Reports" },
  { icon: "users", text: "Team" },
  { icon: "cog", text: "Settings" }
]

for (item in sidebarItems) {
  sidebarItem = Talanolius.HBox.new({
    padding: "12px 20px",
    spacing: 10,
    alignItems: "center",
    backgroundColor: item.active ? "rgba(67, 97, 238, 0.1)" : "transparent",
    color: item.active ? "#4361ee" : "inherit",
    fontWeight: item.active ? "500" : "normal",
    cursor: "pointer",
    onHover: function(hovered) {
      if (!item.active && hovered) {
        this.setStyle({ backgroundColor: "rgba(0, 0, 0, 0.05)" })
      } else if (!item.active) {
        this.setStyle({ backgroundColor: "transparent" })
      }
    },
    onClick: function() {
      # Handle sidebar item click
      setActiveSidebarItem(item.text)
    }
  })
  
  icon = Talanolius.Icon.new({
    name: "fa-" + item.icon,
    width: 20,
    textAlign: "center"
  })
  
  text = Talanolius.Label.new({
    text: item.text
  })
  
  sidebarItem.add(icon)
  sidebarItem.add(text)
  sidebar.add(sidebarItem)
}

# Create the main content area
mainContent = Talanolius.VBox.new({
  padding: 20,
  spacing: 30,
  flex: 1,
  overflowY: "auto"
})

# Project Overview Section
overviewSection = Talanolius.VBox.new({
  spacing: 15
})

overviewTitle = Talanolius.Label.new({
  text: "Project Overview",
  fontSize: 18,
  fontWeight: "600",
  color: "#3a0ca3"
})

# Alert component
alert = Talanolius.HBox.new({
  backgroundColor: "rgba(67, 97, 238, 0.1)",
  borderLeft: "4px solid #4361ee",
  color: "#4361ee",
  padding: 15,
  borderRadius: 4,
  spacing: 10,
  alignItems: "center"
})

alertIcon = Talanolius.Icon.new({
  name: "fa-info-circle"
})

alertText = Talanolius.Label.new({
  text: "Welcome to the Task Manager. You have 3 tasks due today."
})

alert.add(alertIcon)
alert.add(alertText)

# Tabs component
tabs = Talanolius.HBox.new({
  borderBottom: "1px solid #e0e0e0"
})

tabItems = ["All Projects", "Active", "Completed"]

for (i = 0; i < tabItems.length; i++) {
  isActive = i == 0
  tab = Talanolius.Label.new({
    text: tabItems[i],
    padding: "10px 20px",
    borderBottom: isActive ? "2px solid #4361ee" : "2px solid transparent",
    color: isActive ? "#4361ee" : "inherit",
    fontWeight: isActive ? "500" : "normal",
    cursor: "pointer",
    onHover: function(hovered) {
      if (!isActive && hovered) {
        this.setStyle({ backgroundColor: "rgba(0, 0, 0, 0.05)" })
      } else if (!isActive) {
        this.setStyle({ backgroundColor: "transparent" })
      }
    },
    onClick: function() {
      # Handle tab click
      setActiveTab(tabItems[i])
    }
  })
  
  tabs.add(tab)
}

# Project cards
function createProjectCard(title, description, progress, dueDate) {
  card = Talanolius.VBox.new({
    backgroundColor: "#ffffff",
    border: "1px solid #e0e0e0",
    borderRadius: 8,
    padding: 20,
    marginBottom: 20
  })
  
  cardTitle = Talanolius.Label.new({
    text: title,
    fontSize: 16,
    fontWeight: "600",
    marginBottom: 10
  })
  
  cardDescription = Talanolius.Label.new({
    text: description,
    color: "#666666",
    fontSize: 14,
    lineHeight: 1.6
  })
  
  progressBar = Talanolius.ProgressBar.new({
    value: progress,
    height: 8,
    backgroundColor: "#f5f5f5",
    progressColor: progress < 50 ? "#ffbd2e" : "#4361ee",
    borderRadius: 4,
    margin: "10px 0"
  })
  
  progressInfo = Talanolius.HBox.new({
    justifyContent: "space-between",
    alignItems: "center",
    marginTop: 10
  })
  
  progressText = Talanolius.Label.new({
    text: progress + "% Complete",
    fontSize: 12,
    color: "#666666"
  })
  
  dueText = Talanolius.Label.new({
    text: "Due: " + dueDate,
    fontSize: 12,
    color: "#666666"
  })
  
  progressInfo.add(progressText)
  progressInfo.add(dueText)
  
  card.add(cardTitle)
  card.add(cardDescription)
  card.add(progressBar)
  card.add(progressInfo)
  
  return card
}

# Add project cards
projectCard1 = createProjectCard(
  "Website Redesign",
  "Redesign the company website with new branding guidelines.",
  75,
  "Jun 15, 2025"
)

projectCard2 = createProjectCard(
  "Mobile App Development",
  "Develop a mobile app for iOS and Android platforms.",
  30,
  "Aug 20, 2025"
)

# Add components to the overview section
overviewSection.add(overviewTitle)
overviewSection.add(alert)
overviewSection.add(tabs)
overviewSection.add(projectCard1)
overviewSection.add(projectCard2)

# ... (form implementation)

mainContent.add(overviewSection)

layout.add(sidebar)
layout.add(mainContent)

# Set window content
window.setContent(layout)

# Show the window
window.show()

# Start the application
app.run()
