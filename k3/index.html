fibonacci.k3 
// Recursive Fibonacci
fn fibonacci_recursive(n: int) -> int {
    if n <= 1 {
        return n
    }
    return fibonacci_recursive(n - 1) + fibonacci_recursive(n - 2)
}

// Iterative Fibonacci
fn fibonacci_iterative(n: int) -> int {
    if n <= 1 {
        return n
    }
    
    let a = 0
    let b = 1
    
    for i in 2..=n {
        let temp = a + b
        a = b
        b = temp
    }
    
    return b
}

// Memoized Fibonacci
fn fibonacci_memoized(n: int) -> int {
    static memo: Map<int, int> = {}
    
    if memo.has(n) {
        return memo.get(n)
    }
    
    if n <= 1 {
        return n
    }
    
    let result = fibonacci_memoized(n - 1) + fibonacci_memoized(n - 2)
    memo.set(n, result)
    
    return result
}

fn main() {
    let n = 40
    
    let start = Time.now()
    let result1 = fibonacci_recursive(n)
    let end = Time.now()
    println("Recursive: {} (took {} ms)", result1, end - start)
    
    start = Time.now()
    let result2 = fibonacci_iterative(n)
    end = Time.now()
    println("Iterative: {} (took {} ms)", result2, end - start)
    
    start = Time.now()
    let result3 = fibonacci_memoized(n)
    end = Time.now()
    println("Memoized: {} (took {} ms)", result3, end - start)
}

import Http from "k3/http"
import Router from "k3/http/router"
import { readFile } from "k3/fs"

// Create a new router
let router = Router.new()

// Define routes
router.get("/", async (req, res) => {
    let content = await readFile("public/index.html", "utf-8")
    res.setHeader("Content-Type", "text/html")
    res.write(content)
    res.end()
})

router.get("/api/users", async (req, res) => {
    let users = [
        { id: 1, name: "Alice" },
        { id: 2, name: "Bob" },
        { id: 3, name: "Charlie" }
    ]
    
    res.setHeader("Content-Type", "application/json")
    res.write(JSON.stringify(users))
    res.end()
})

router.post("/api/users", async (req, res) => {
    let body = await req.json()
    
    // Validate input
    if (!body.name) {
        res.statusCode = 400
        res.write(JSON.stringify({ error: "Name is required" }))
        res.end()
        return
    }
    
    // In a real app, you would save to a database
    res.statusCode = 201
    res.setHeader("Content-Type", "application/json")
    res.write(JSON.stringify({ id: 4, name: body.name }))
    res.end()
})

// Create and start the server
fn main() {
    let server = Http.createServer((req, res) => {
        router.handle(req, res)
    })
    
    server.listen(3000, () => {
        println("Server running at http://localhost:3000")
    })
}

counter_app.k3 
import Stellar from "stellar/ui"

// A simple counter component
class Counter extends Stellar.Component {
    state = {
        count: 0
    }
    
    fn increment() {
        this.setState({ count: this.state.count + 1 })
    }
    
    fn decrement() {
        this.setState({ count: this.state.count - 1 })
    }
    
    fn render() -> Stellar.Element {
        return <div className="counter">
            <h1>Counter: {this.state.count}</h1>
            <div className="buttons">
                <button onClick={this.decrement}>-</button>
                <button onClick={this.increment}>+</button>
            </div>
        </div>
    }
}

// A todo list component
class TodoList extends Stellar.Component {
    state = {
        todos: [],
        newTodo: ""
    }
    
    fn handleInput(event) {
        this.setState({ newTodo: event.target.value })
    }
    
    fn addTodo() {
        if (this.state.newTodo.trim() == "") {
            return
        }
        
        let newTodos = [...this.state.todos, {
            id: Date.now(),
            text: this.state.newTodo,
            completed: false
        }]
        
        this.setState({
            todos: newTodos,
            newTodo: ""
        })
    }
    
    fn toggleTodo(id) {
        let newTodos = this.state.todos.map(todo => {
            if (todo.id == id) {
                return { ...todo, completed: !todo.completed }
            }
            return todo
        })
        
        this.setState({ todos: newTodos })
    }
    
    fn render() -> Stellar.Element {
        return <div className="todo-list">
            <h1>Todo List</h1>
            
            <div className="add-todo">
                <input 
                    type="text" 
                    value={this.state.newTodo}
                    onChange={this.handleInput}
                    placeholder="Add a new todo"
                />
                <button onClick={this.addTodo}>Add</button>
            </div>
            
            <ul>
                {this.state.todos.map(todo => (
                    <li 
                        key={todo.id}
                        className={todo.completed ? "completed" : ""}
                        onClick={() => this.toggleTodo(todo.id)}
                    >
                        {todo.text}
                    </li>
                ))}
            </ul>
        </div>
    }
}

// Main app component
class App extends Stellar.Component {
    fn render() -> Stellar.Element {
        return <div className="app">
            <Counter />
            <TodoList />
        </div>
    }
}

// Start the UI application
fn main() {
    Stellar.render(<App />, "#app")
}

concurrency.k3 
import { sleep } from "k3/time"
import { spawn, Channel } from "k3/concurrency"

// Async/await example
async fn fetchUserData(userId: int) -> Result<User, Error> {
    println("Fetching user data for user {}", userId)
    
    // Simulate network request
    await sleep(1000)
    
    // Simulate successful response
    return Ok(User {
        id: userId,
        name: "User " + userId.toString(),
        email: "user" + userId.toString() + "@example.com"
    })
}

// Parallel data fetching
async fn fetchAllUsers(userIds: []int) -> []User {
    // Create an array of promises
    let promises = userIds.map(id => fetchUserData(id))
    
    // Wait for all promises to resolve
    let results = await Promise.all(promises)
    
    // Filter out errors and return successful results
    return results
        .filter(result => result.isOk())
        .map(result => result.unwrap())
}

// Actor model example
actor Counter {
    count: int = 0
    
    fn increment() {
        this.count += 1
        return this.count
    }
    
    fn decrement() {
        this.count -= 1
        return this.count
    }
    
    fn getCount() {
        return this.count
    }
}

// Channels and lightweight threads
fn producer(channel: Channel<int>, count: int) {
    for i in 0..count {
        println("Producing: {}", i)
        channel.send(i)
        sleep(100)
    }
    channel.close()
}

fn consumer(channel: Channel<int>, id: int) {
    while true {
        match channel.receive() {
            Some(value) => println("Consumer {}: Received {}", id, value),
            None => break // Channel closed
        }
    }
}

fn main() {
    // Async/await example
    spawn(async () => {
        let users = await fetchAllUsers([1, 2, 3, 4, 5])
        println("Fetched {} users", users.length)
        
        for user in users {
            println("User: {}", user.name)
        }
    })
    
    // Actor example
    let counter = Counter.new()
    
    for i in 0..5 {
        spawn(async () => {
            for j in 0..10 {
                let value = await counter.increment()
                println("Actor thread {}: Counter = {}", i, value)
                await sleep(100)
            }
        })
    }
    
    // Channels example
    let channel = Channel<int>.new(5) // Buffered channel with capacity 5
    
    spawn(() => producer(channel, 10))
    
    for i in 0..3 {
        spawn(() => consumer(channel, i))
    }
    
    // Wait for all tasks to complete
    sleep(3000)
}

todo_app.k3 
import Stellar from "stellar/ui"

class TodoApp extends Stellar.Component {
    state = {
        todos: [],
        newTodo: ""
    }
    
    fn handleInput(event) {
        this.setState({ newTodo: event.target.value })
    }
    
    fn addTodo() {
        if (this.state.newTodo.trim() == "") {
            return
        }
        
        let newTodos = [...this.state.todos, {
            id: Date.now(),
            text: this.state.newTodo,
            completed: false
        }]
        
        this.setState({
            todos: newTodos,
            newTodo: ""
        })
    }
    
    fn toggleTodo(id) {
        let newTodos = this.state.todos.map(todo => {
            if (todo.id == id) {
                return { ...todo, completed: !todo.completed }
            }
            return todo
        })
        
        this.setState({ todos: newTodos })
    }
    
    fn deleteTodo(id) {
        let newTodos = this.state.todos.filter(todo => todo.id != id)
        this.setState({ todos: newTodos })
    }
    
    fn render() -> Stellar.Element {
        return <div className="todo-app">
            <h1>Todo List</h1>
            
            <div className="add-todo">
                <input 
                    type="text" 
                    value={this.state.newTodo}
                    onChange={this.handleInput}
                    onKeyPress={e => e.key == "Enter" && this.addTodo()}
                    placeholder="Add a new todo"
                />
                <button onClick={this.addTodo}>Add</button>
            </div>
            
            <ul className="todo-list">
                {this.state.todos.length == 0 ? (
                    <li className="empty-state">No todos yet. Add one above!</li>
                ) : (
                    this.state.todos.map(todo => (
                        <li key={todo.id} className={todo.completed ? "completed" : ""}>
                            <input 
                                type="checkbox" 
                                checked={todo.completed}
                                onChange={() => this.toggleTodo(todo.id)}
                            />
                            <span>{todo.text}</span>
                            <button 
                                className="delete-btn"
                                onClick={() => this.deleteTodo(todo.id)}
                            >
                                Delete
                            </button>
                        </li>
                    ))
                )}
            </ul>
            
            {this.state.todos.length > 0 && (
                <div className="todo-stats">
                    {this.state.todos.filter(t => !t.completed).length} items left
                </div>
            )}
        </div>
    }
}

fn main() {
    Stellar.render(<TodoApp />, "#app")
}

Installation Instructions

  1. Download the Windows installer (.exe)
  2. Run the installer and follow the prompts
  3. Open Command Prompt and type k3 --version to verify installation
  1. Download the macOS package (.pkg)
  2. Open the package and follow the installation wizard
  3. Open Terminal and type k3 --version to verify installation
  1. Download the Linux tarball (.tar.gz)
  2. Extract it: tar -xzf k3-1.0.0-linux-x64.tar.gz
  3. Move to a directory in your PATH: sudo mv k3-1.0.0 /usr/local/k3
  4. Create a symlink: sudo ln -s /usr/local/k3/bin/k3 /usr/local/bin/k3
  5. Verify installation: k3 --version

concurrency.k3 
import { sleep } from "k3/time"
import { spawn, Channel } from "k3/concurrency"

// Async/await example
async fn fetchUserData(userId: int) -> Result<User, Error> {
    println("Fetching user data for user {}", userId)
    
    // Simulate network request
    await sleep(1000)
    
    // Simulate successful response
    return Ok(User {
        id: userId,
        name: "User " + userId.toString(),
        email: "user" + userId.toString() + "@example.com"
    })
}

// Parallel data fetching
async fn fetchAllUsers(userIds: []int) -> []User {
    // Create an array of promises
    let promises = userIds.map(id => fetchUserData(id))
    
    // Wait for all promises to resolve
    let results = await Promise.all(promises)
    
    // Filter out errors and return successful results
    return results
        .filter(result => result.isOk())
        .map(result => result.unwrap())
}

// Actor model example
actor Counter {
    count: int = 0
    
    fn increment() {
        this.count += 1
        return this.count
    }
    
    fn decrement() {
        this.count -= 1
        return this.count
    }
    
    fn getCount() {
        return this.count
    }
}

// Channels and lightweight threads
fn producer(channel: Channel<int>, count: int) {
    for i in 0..count {
        println("Producing: {}", i)
        channel.send(i)
        sleep(100)
    }
    channel.close()
}

fn consumer(channel: Channel<int>, id: int) {
    while true {
        match channel.receive() {
            Some(value) => println("Consumer {}: Received {}", id, value),
            None => break // Channel closed
        }
    }
}

fn main() {
    // Async/await example
    spawn(async () => {
        let users = await fetchAllUsers([1, 2, 3, 4, 5])
        println("Fetched {} users", users.length)
        
        for user in users {
            println("User: {}", user.name)
        }
    })
    
    // Actor example
    let counter = Counter.new()
    
    for i in 0..5 {
        spawn(async () => {
            for j in 0..10 {
                let value = await counter.increment()
                println("Actor thread {}: Counter = {}", i, value)
                await sleep(100)
            }
        })
    }
    
    // Channels example
    let channel = Channel<int>.new(5) // Buffered channel with capacity 5
    
    spawn(() => producer(channel, 10))
    
    for i in 0..3 {
        spawn(() => consumer(channel, i))
    }
    
    // Wait for all tasks to complete
    sleep(3000)
}

todo_app.k3 
import Stellar from "stellar/ui"

class TodoApp extends Stellar.Component {
    state = {
        todos: [],
        newTodo: ""
    }
    
    fn handleInput(event) {
        this.setState({ newTodo: event.target.value })
    }
    
    fn addTodo() {
        if (this.state.newTodo.trim() == "") {
            return
        }
        
        let newTodos = [...this.state.todos, {
            id: Date.now(),
            text: this.state.newTodo,
            completed: false
        }]
        
        this.setState({
            todos: newTodos,
            newTodo: ""
        })
    }
    
    fn toggleTodo(id) {
        let newTodos = this.state.todos.map(todo => {
            if (todo.id == id) {
                return { ...todo, completed: !todo.completed }
            }
            return todo
        })
        
        this.setState({ todos: newTodos })
    }
    
    fn deleteTodo(id) {
        let newTodos = this.state.todos.filter(todo => todo.id != id)
        this.setState({ todos: newTodos })
    }
    
    fn render() -> Stellar.Element {
        return <div className="todo-app">
            <h1>Todo List</h1>
            
            <div className="add-todo">
                <input 
                    type="text" 
                    value={this.state.newTodo}
                    onChange={this.handleInput}
                    onKeyPress={e => e.key == "Enter" && this.addTodo()}
                    placeholder="Add a new todo"
                />
                <button onClick={this.addTodo}>Add</button>
            </div>
            
            <ul className="todo-list">
                {this.state.todos.length == 0 ? (
                    <li className="empty-state">No todos yet. Add one above!</li>
                ) : (
                    this.state.todos.map(todo => (
                        <li key={todo.id} className={todo.completed ? "completed" : ""}>
                            <input 
                                type="checkbox" 
                                checked={todo.completed}
                                onChange={() => this.toggleTodo(todo.id)}
                            />
                            <span>{todo.text}</span>
                            <button 
                                className="delete-btn"
                                onClick={() => this.deleteTodo(todo.id)}
                            >
                                Delete
                            </button>
                        </li>
                    ))
                )}
            </ul>
            
            {this.state.todos.length > 0 && (
                <div className="todo-stats">
                    {this.state.todos.filter(t => !t.completed).length} items left
                </div>
            )}
        </div>
    }
}

fn main() {
    Stellar.render(<TodoApp />, "#app")
}
